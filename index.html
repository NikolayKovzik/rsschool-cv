  function camelize (str) {
    let arrWithoutDash = str.split("-");
    
    arrWithoutDash.forEach((item,index,array)=> {
      if(index !== 0 && item!=='') {
          array[index] = array[index][0].toUpperCase(); 
          array[index] += array[index].slice(1);
      }
    });
    
    let result = arrWithoutDash.filter((item)=>{
      if(item === "")
        return false;
        else return true;
    });
    return result.join('');
    }
