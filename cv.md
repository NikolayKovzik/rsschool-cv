function sortByBit(arr) {
    let bitForm = {};
    arr.forEach((item) => {
        let bits = item.toString(2).match(/1/g);
        bitForm[item] = bits ? bits.length : 0
    })

    return arr.sort((a, b) => {
        return (bitForm[a] === bitForm[b]) ? a - b : bitForm[a] - bitForm[b]
    });
}
